#!/bin/bash

#inod_snmp all open files with tupe "CHR" in all squid process

 

declare -a PID_ar

nn_p=squid

# заносис идентификаторы сквида в массив

PID_ar=( `ps aux|grep $nn_p|grep -v "grep"|awk '{print $2}'`)

 

if [[ $PID_ar ]] ; then

#echo ${PID_ar[@]}

INOD_AV=`squidclient mgr:info | grep -e 'A.*file descri.*' | grep -o -E '[0-9]*'`

echo $INOD_AV

INOD_CAL=0

for (( i=0; i < ${#PID_ar[@]} ; i++ )) #Собираем открытые фалы со всех процессов

do

 

let "INOD_CAL=$INOD_CAL+`lsof -p ${PID_ar[$i]} | awk '{ print $5, $9}' | sed -n '/^CHR.*$/p' | wc -l`"

#слаживаем количество открытых файлов с типом CHR

done

echo $INOD_CAL

 

else echo "сквид не запущен"

 

fi
